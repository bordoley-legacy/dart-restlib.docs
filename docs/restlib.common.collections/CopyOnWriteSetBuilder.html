class CopyOnWriteSetBuilder<E> implements CopyOnWriteCollectionBuilder<E> {

     _set.add(element);
 
 void addAll(final Iterable<E> elements) =>
     _set.addAll(elements);
 
 ImmutableSet<E> build() => 
     new _CopyOnWriteSet(
         new MutableSet.hash(elements: _set));
