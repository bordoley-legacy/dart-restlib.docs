final Parser<Path> PATH # 

final Parser<Path> PATH = 
 _PATH_SEGMENT
   .sepBy(FORWARD_SLASH)
   .map(Path.EMPTY.addAll)
   .map((final Path path) {
     if ((path.length == 1) && path.elementAt(0).isEmpty) {
       return Path.EMPTY;
     } else {
       return path;
     }
   })
